const concept = ['arrays', 'can', 'be', 'mutated'];

function changeArr(arr){
  arr[3] = 'MUTATED';//создаем функцию кот изменяет 3 эллемент
}
changeArr(concept);
console.log(concept);

function removeElement(newArr){
  newArr.pop();//эта функция удаляет последний эллемент
}
removeElement(concept);
console.log(concept);
